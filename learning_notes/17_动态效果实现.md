## 粒子系统

### 粒子系统概述

粒子系统用于创建各种动态视觉效果：

| 效果类型 | 应用场景 |
|----------|----------|
| 星星闪烁 | 开心、兴奋 |
| 爱心飘落 | 浪漫特效 |
| 火花飞溅 | 点击反馈 |
| 烟雾效果 | 梦境、魔法 |
| 雨滴下落 | 天气效果 |

### 粒子类结构

```python
class Particle:
    def __init__(self):
        self.x = 0.0
        self.y = 0.0
        self.vx = 0.0
        self.vy = 0.0
        self.life = 1.0
        self.max_life = 1.0
        self.size = 5.0
        self.color = (255, 255, 255, 255)
        self.rotation = 0.0
        self.rotation_speed = 0.0

    def update(self, delta_time):
        self.x += self.vx * delta_time
        self.y += self.vy * delta_time
        self.life -= delta_time
        self.rotation += self.rotation_speed * delta_time

    @property
    def alpha(self):
        return max(0, int(255 * (self.life / self.max_life)))

    @property
    def is_alive(self):
        return self.life > 0
```

### 粒子发射器

```python
class ParticleEmitter:
    def __init__(self):
        self.particles = []
        self.emission_rate = 10  # 每秒发射粒子数
        self.emission_timer = 0.0
        self.max_particles = 100
        self.position = (0, 0)
        self.spread = 0.5  # 发射扩散范围
        self.particle_lifetime = 2.0
        self.particle_speed = (50, 100)  # (min, max)
        self.particle_size_range = (2, 8)

    def emit(self, count=1):
        """发射粒子"""
        for _ in range(count):
            if len(self.particles) >= self.max_particles:
                break

            particle = Particle()
            particle.x = self.position[0] + random.uniform(-self.spread, self.spread)
            particle.y = self.position[1] + random.uniform(-self.spread, self.spread)

            speed = random.uniform(*self.particle_speed)
            angle = random.uniform(0, 2 * math.pi)
            particle.vx = math.cos(angle) * speed
            particle.vy = math.sin(angle) * speed - 50  # 向上偏移

            particle.life = self.particle_lifetime
            particle.max_life = self.particle_lifetime
            particle.size = random.uniform(*self.particle_size_range)

            self.particles.append(particle)

    def update(self, delta_time):
        # 发射新粒子
        self.emission_timer += delta_time
        emit_interval = 1.0 / self.emission_rate

        while self.emission_timer >= emit_interval:
            self.emit()
            self.emission_timer -= emit_interval

        # 更新现有粒子
        for particle in self.particles[:]:
            particle.update(delta_time)
            if not particle.is_alive:
                self.particles.remove(particle)

    def render(self, painter):
        for particle in self.particles:
            color = QColor(*particle.color)
            color.setAlpha(particle.alpha)

            painter.setOpacity(particle.alpha / 255)
            painter.setBrush(color)
            painter.drawEllipse(
                particle.x - particle.size / 2,
                particle.y - particle.size / 2,
                particle.size,
                particle.size
            )
```

### 粒子效果类型

#### 星星闪烁效果

```python
class SparkleEmitter(ParticleEmitter):
    def __init__(self):
        super().__init__()
        self.emission_rate = 5
        self.particle_lifetime = 1.5
        self.particle_speed = (20, 50)
        self.colors = [
            (255, 215, 0, 255),   # 金色
            (255, 255, 255, 255), # 白色
            (255, 182, 193, 255), # 粉色
        ]

    def emit(self, count=1):
        for _ in range(count):
            particle = Particle()
            particle.x = self.position[0] + random.uniform(-50, 50)
            particle.y = self.position[1] + random.uniform(-50, 50)

            speed = random.uniform(*self.particle_speed)
            angle = -math.pi / 2 + random.uniform(-0.5, 0.5)  # 向上
            particle.vx = math.cos(angle) * speed
            particle.vy = math.sin(angle) * speed

            particle.life = self.particle_lifetime
            particle.max_life = self.particle_lifetime
            particle.size = random.uniform(3, 8)
            particle.color = random.choice(self.colors)
            particle.rotation_speed = random.uniform(-5, 5)

            self.particles.append(particle)

    def render(self, painter):
        for particle in self.particles:
            painter.save()
            painter.translate(particle.x, particle.y)
            painter.rotate(particle.rotation)

            # 绘制四角星
            size = particle.size * (0.5 + particle.alpha / 500)
            painter.setOpacity(particle.alpha / 255)

            color = QColor(*particle.color)
            color.setAlpha(particle.alpha)
            painter.setBrush(color)

            # 绘制星形
            self.draw_star(painter, 0, 0, size)

            painter.restore()

    def draw_star(self, painter, x, y, size):
        """绘制四角星"""
        path = QPainterPath()
        points = 4
        outer_radius = size
        inner_radius = size * 0.4

        for i in range(points * 2):
            radius = outer_radius if i % 2 == 0 else inner_radius
            angle = (i * math.pi / points) - math.pi / 2

            px = x + math.cos(angle) * radius
            py = y + math.sin(angle) * radius

            if i == 0:
                path.moveTo(px, py)
            else:
                path.lineTo(px, py)

        path.closeSubpath()
        painter.drawPath(path)
```

#### 点击涟漪效果

```python
class RippleEmitter(ParticleEmitter):
    def __init__(self):
        super().__init__()
        self.max_particles = 20
        self.ripple_color = (100, 149, 237, 200)  # 蓝色

    def emit_at(self, x, y):
        """在指定位置发射涟漪"""
        self.position = (x, y)

        for i in range(10):
            particle = Particle()
            particle.x = x
            particle.y = y

            angle = (i / 10) * 2 * math.pi
            speed = random.uniform(30, 60)
            particle.vx = math.cos(angle) * speed
            particle.vy = math.sin(angle) * speed

            particle.life = 0.8
            particle.max_life = 0.8
            particle.size = random.uniform(2, 4)

            self.particles.append(particle)

    def render(self, painter):
        for particle in self.particles:
            color = QColor(*self.ripple_color)
            color.setAlpha(particle.alpha)

            painter.setOpacity(particle.alpha / 255)
            painter.setPen(color)
            painter.setBrush(Qt.NoBrush)

            # 绘制圆环
            radius = (1.0 - particle.life / particle.max_life) * 50
            painter.drawEllipse(
                particle.x - radius,
                particle.y - radius,
                radius * 2,
                radius * 2
            )
```

## 特效渲染

### 特效基础

```python
class Effect:
    def __init__(self):
        self.enabled = True
        self.duration = 1.0
        self.elapsed = 0.0
        self.loop = False

    def update(self, delta_time):
        self.elapsed += delta_time
        if self.elapsed >= self.duration and not self.loop:
            self.enabled = False

    @property
    def progress(self):
        return min(1.0, self.elapsed / self.duration) if self.duration > 0 else 1.0
```

### 渐变效果

```python
class FadeEffect(Effect):
    def __init__(self, fade_in=True, duration=0.5):
        super().__init__()
        self.fade_in = fade_in
        self.duration = duration

    def get_opacity(self):
        if self.fade_in:
            return self.progress
        else:
            return 1.0 - self.progress
```

### 缩放效果

```python
class ScaleEffect(Effect):
    def __init__(self, from_scale=0.0, to_scale=1.0, duration=0.3):
        super().__init__()
        self.from_scale = from_scale
        self.to_scale = to_scale
        self.duration = duration

    def get_scale(self):
        t = self.ease_out_back(self.progress)
        return self.from_scale + (self.to_scale - self.from_scale) * t

    @staticmethod
    def ease_out_back(t):
        """带超调的缓动函数"""
        c1 = 1.70158
        c3 = c1 + 1
        return 1 + c3 * pow(t - 1, 3) + c1 * pow(t - 1, 2)
```

## 高级视觉效果

### 辉光效果

```python
class GlowEffect:
    def __init__(self, color=(255, 255, 200), blur_radius=20):
        self.color = color
        self.blur_radius = blur_radius

    def apply(self, painter, rect):
        """在绘制角色后应用辉光"""
        glow_rect = rect.adjusted(
            -self.blur_radius,
            -self.blur_radius,
            self.blur_radius,
            self.blur_radius
        )

        # 创建渐变
        gradient = QRadialGradient(
            glow_rect.center(),
            max(glow_rect.width(), glow_rect.height()) / 2
        )

        color = QColor(*self.color)
        gradient.setColorAt(0, color.lighter(200))
        gradient.setColorAt(0.5, color.lighter(150))
        gradient.setColorAt(1, QColor(0, 0, 0, 0))

        painter.fillRect(glow_rect, gradient)
```

### 马赛克效果

```python
class MosaicEffect:
    def __init__(self, grid_size=10):
        self.grid_size = grid_size

    def apply(self, image):
        """对图像应用马赛克"""
        width = image.width()
        height = image.height()

        for x in range(0, width, self.grid_size):
            for y in range(0, height, self.grid_size):
                # 获取区块平均颜色
                avg_color = self.get_average_color(image, x, y)

                # 应用到整个区块
                for sx in range(x, min(x + self.grid_size, width)):
                    for sy in range(y, min(y + self.grid_size, height)):
                        image.setPixel(sx, sy, avg_color)

        return image

    def get_average_color(self, image, x, y):
        r, g, b, a = 0, 0, 0, 0
        count = 0

        for sx in range(x, min(x + self.grid_size, image.width())):
            for sy in range(y, min(y + self.grid_size, image.height())):
                color = image.pixel(sx, sy)
                r += qRed(color)
                g += qGreen(color)
                b += qBlue(color)
                a += qAlpha(color)
                count += 1

        if count > 0:
            return qRgba(r // count, g // count, b // count, a // count)
        return 0
```

### 动态模糊

```python
class MotionBlur:
    def __init__(self, strength=5):
        self.strength = strength
        self.frames = []

    def add_frame(self, frame):
        self.frames.append(frame)
        if len(self.frames) > self.strength:
            self.frames.pop(0)

    def get_blurred_frame(self):
        if not self.frames:
            return None

        # 混合所有帧
        result = self.frames[0].copy()
        for frame in self.frames[1:]:
            result = self.blend_frames(result, frame, 0.3)

        return result

    def blend_frames(self, frame1, frame2, alpha):
        # 简单的帧混合
        painter = QPainter(frame1)
        painter.setOpacity(alpha)
        painter.drawImage(0, 0, frame2)
        painter.end()
        return frame1
```

## 性能优化

### 粒子数量限制

```python
class ParticleSystem:
    def __init__(self, max_particles=500):
        self.max_particles = max_particles
        self.emitters = []
        self.enabled = True

    def update(self, delta_time):
        if not self.enabled:
            return

        total_particles = sum(len(e.particles) for e in self.emitters)

        for emitter in self.emitters:
            # 如果总粒子数过多，临时降低发射率
            if total_particles > self.max_particles:
                original_rate = emitter.emission_rate
                emitter.emission_rate *= 0.5
                emitter.update(delta_time)
                emitter.emission_rate = original_rate
            else:
                emitter.update(delta_time)
```

### 渲染批处理

```python
class BatchedRenderer:
    def __init__(self):
        self.batch_operations = []

    def add_to_batch(self, operation):
        self.batch_operations.append(operation)

    def execute_batch(self, painter):
        """批量执行渲染操作"""
        # 按类型分组
        grouped = defaultdict(list)
        for op in self.batch_operations:
            grouped[op['type']].append(op)

        # 分组渲染
        for op_type, operations in grouped.items():
            if op_type == 'circle':
                self.render_circles(painter, operations)
            elif op_type == 'rect':
                self.render_rects(painter, operations)
            elif op_type == 'image':
                self.render_images(painter, operations)

        self.batch_operations.clear()

    def render_circles(self, painter, operations):
        for op in operations:
            painter.setBrush(op['color'])
            painter.drawEllipse(op['rect'])

    def render_rects(self, painter, operations):
        for op in operations:
            painter.fillRect(op['rect'], op['color'])
```

## 项目中的效果实现

### 项目中的点击效果

项目中的点击效果实现位于 `src/ui/desktop_pet_widget.py`：

```python
class DesktopPetWidget:
    def __init__(self):
        self.effects = []
        self.particle_system = ParticleSystem()

    def on_click(self, event):
        # 获取点击位置
        click_pos = event.pos()

        # 创建涟漪效果
        self.create_ripple(click_pos)

        # 创建星星效果
        self.create_sparkle(click_pos)

    def create_ripple(self, pos):
        ripple = RippleEmitter()
        ripple.emit_at(pos.x(), pos.y())
        self.particle_system.add_emitter(ripple)

    def create_sparkle(self, pos):
        sparkle = SparkleEmitter()
        sparkle.position = (pos.x(), pos.y())
        sparkle.emit(5)
        self.particle_system.add_emitter(sparkle)
```

### 表情切换效果

```python
class ExpressionTransition:
    def __init__(self):
        self.current_effect = None

    def start_transition(self, expression_type):
        if expression_type == "Happy":
            self.current_effect = ScaleEffect(0.8, 1.2, 0.3)
            self.current_effect.next_effect = ScaleEffect(1.2, 1.0, 0.2)
        elif expression_type == "Surprised":
            self.current_effect = FadeEffect(fade_in=False, duration=0.1)
            self.current_effect.next_effect = FadeEffect(fade_in=True, duration=0.2)

    def update(self, delta_time):
        if self.current_effect:
            self.current_effect.update(delta_time)
            if not self.current_effect.enabled and hasattr(self.current_effect, 'next_effect'):
                self.current_effect = self.current_effect.next_effect
```
