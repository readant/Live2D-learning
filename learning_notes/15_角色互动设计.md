## 交互设计基础

### 交互设计原则

| 原则 | 说明 | 实现方法 |
|------|------|----------|
| 即时反馈 | 操作后立即响应 | 快速动画、声音反馈 |

| 可预期性 | 结果可预测 | 明确的交互暗示 |

### 项目中的鼠标交互实现

```python
class InteractionController:
    def __init__(self):
        self.click_count = 0
        self.click_timer = 0.0
        self.click_interval = 0.3  # 300ms 内算连击
        self.last_click_pos = None
        self.drag_mode = False

    def on_mouse_press(self, x, y):
        self.click_count += 1
        self.click_timer = self.click_interval
        self.last_click_pos = (x, y)

        # 判断点击区域
        if self.is_head_click(x, y):
            self.handle_head_click()
        else:
            self.handle_body_click()

    def on_mouse_release(self, x, y):
        if self.drag_mode:
            self.end_drag()
        elif self.is_flick(x, y):
            self.handle_flick()

    def on_mouse_move(self, x, y):
        if self.drag_mode:
            self.update_drag(x, y)
        else:
            self.update_gaze(x, y)

    def update(self, delta_time):
        # 更新连击计时器
        if self.click_timer > 0:
            self.click_timer -= delta_time
            if self.click_timer <= 0:
                self.click_count = 0
```

```python
class VoiceInteraction:
    def __init__(self):
        self.voice_enabled = False
        self.voice_recognition = None

    def start_listening(self):
        # 启动语音识别
        self.voice_recognition.start()

    def on_voice_input(self, text):
        # 分析语音输入
        intent = self.analyze_intent(text)

        if intent == "greeting":
            self.play_greeting_response()
        elif intent == "question":
            self.play_thinking_response()
        elif intent == "command":
            self.execute_command(text)

    def play_greeting_response(self):
        # 播放问候动作和表情
        self.animation_controller.play("Wave")
        self.expression_controller.switch("Happy")
```

## 反馈系统

### 视觉反馈

```python
class VisualFeedback:
    def __init__(self):
        self.highlight_effect = False
        self.click_effect = False
        self.particle_system = ParticleSystem()

    def on_hover_start(self):
        # 悬停开始 - 高亮效果
        self.highlight_effect = True
        self.model.set_opacity(1.0)

    def on_hover_end(self):
        # 悬停结束 - 恢复正常
        self.highlight_effect = False

    def on_click(self, pos):
        # 点击效果 - 涟漪粒子
        self.particle_system.emit(pos, "ripple", count=10)

    def update(self, delta_time):
        self.particle_system.update(delta_time)
```

### 声音反馈

```python
import pygame

class AudioFeedback:
    def __init__(self):
        pygame.mixer.init()
        self.sounds = {
            'click': pygame.mixer.Sound('sounds/click.wav'),
            'hover': pygame.mixer.Sound('sounds/hover.wav'),
            'success': pygame.mixer.Sound('sounds/success.wav'),
        }
        self.volume = 0.5

    def play_sound(self, sound_name):
        if sound_name in self.sounds:
            sound = self.sounds[sound_name]
            sound.set_volume(self.volume)
            sound.play()

    def play_click(self):
        self.play_sound('click')

    def play_hover(self):
        self.play_sound('hover')
```

## 情感交互

### 情感状态系统

```python
class EmotionState:
    NEUTRAL = "neutral"
    HAPPY = "happy"
    SAD = "sad"
    ANGRY = "angry"
    SURPRISED = "surprised"
    EXCITED = "excited"

class EmotionalInteraction:
    def __init__(self):
        self.current_emotion = EmotionState.NEUTRAL
        self.emotion_intensity = 0.5
        self.emotion_decay = 0.1
        self.memory = EmotionMemory()

    def on_positive_interaction(self):
        """正面互动"""
        self.current_emotion = EmotionState.HAPPY
        self.emotion_intensity = min(1.0, self.emotion_intensity + 0.2)
        self.memory.add_interaction("positive")

    def on_negative_interaction(self):
        """负面互动"""
        self.current_emotion = EmotionState.SAD
        self.emotion_intensity = min(1.0, self.emotion_intensity + 0.2)
        self.memory.add_interaction("negative")

    def on_exciting_event(self):
        """兴奋事件"""
        self.current_emotion = EmotionState.EXCITED
        self.emotion_intensity = 1.0
        self.play_excited_animation()

    def update(self, delta_time):
        # 情感衰减
        if self.emotion_intensity > 0.5:
            self.emotion_intensity -= self.emotion_decay * delta_time

        # 更新表情和动作
        self.apply_emotion_to_model()

    def apply_emotion_to_model(self):
        emotion_config = EMOTION_CONFIGS[self.current_emotion]
        blended = self.blend_emotion(emotion_config, self.emotion_intensity)
        self.apply_to_model(blended)
```

### 情感记忆

```python
class EmotionMemory:
    def __init__(self):
        self.interactions = []
        self.max_memory = 50

    def add_interaction(self, interaction_type):
        self.interactions.append({
            'type': interaction_type,
            'timestamp': time.time()
        })

        # 限制记忆长度
        if len(self.interactions) > self.max_memory:
            self.interactions.pop(0)

    def get_mood_trend(self):
        """计算情绪趋势"""
        recent = self.interactions[-10:]  # 最近10次
        positive = sum(1 for i in recent if i['type'] == 'positive')
        negative = sum(1 for i in recent if i['type'] == 'negative')

        if positive > negative:
            return "positive"
        elif negative > positive:
            return "negative"
        else:
            return "neutral"
```

## 交互状态机

### 状态定义

```python
class CharacterState(Enum):
    IDLE = "idle"              # 待机
    AWARE = "aware"           # 感知（发现用户）
    INTERACTING = "interacting"  # 交互中
    EXCITED = "excited"       # 兴奋
    TIRED = "tired"           # 疲劳
    SLEEPING = "sleeping"     # 睡眠
    LEAVING = "leaving"        # 离开
```

### 状态转换图

```
                    用户离开
    ┌──────────────┐    ┌──────────────┐
    │              │    │              │
    ▼              │    ▼              │
┌──────┐      ┌────────┐         ┌────────┐
│ IDLE │ ←──→ │ AWARE  │ ←──────→ │INTERACT│
└──────┘      └────────┘          └────────┘
    │              │                  │
    │ 用户长时间    │                  │ 连续互动
    │ 无操作        │                  │
    ▼              │                  ▼
┌──────┐      ┌────────┐         ┌────────┐
│SLEEP │      │ TIRED  │         │EXCITED │
└──────┘      └────────┘         └────────┘
```

### 状态机实现

```python
class InteractionStateMachine:
    def __init__(self):
        self.current_state = CharacterState.IDLE
        self.state_timer = 0.0
        self.state_handlers = {
            CharacterState.IDLE: self.handle_idle,
            CharacterState.AWARE: self.handle_aware,
            CharacterState.INTERACTING: self.handle_interacting,
            CharacterState.EXCITED: self.handle_excited,
            CharacterState.TIRED: self.handle_tired,
            CharacterState.SLEEPING: self.handle_sleeping,
        }

    def update(self, delta_time):
        self.state_timer += delta_time

        # 执行当前状态的处理
        handler = self.state_handlers[self.current_state]
        handler(delta_time)

        # 检查状态转换
        self.check_transitions()

    def check_transitions(self):
        new_state = None

        if self.current_state == CharacterState.IDLE:
            if self.detect_user_presence():
                new_state = CharacterState.AWARE
            elif self.state_timer > 300:  # 5分钟无操作
                new_state = CharacterState.SLEEPING

        elif self.current_state == CharacterState.AWARE:
            if self.user_is_interacting():
                new_state = CharacterState.INTERACTING
            elif self.state_timer > 10:
                new_state = CharacterState.IDLE

        elif self.current_state == CharacterState.INTERACTING:
            if self.get_interaction_count() > 10:
                new_state = CharacterState.EXCITED
            elif not self.user_is_interacting():
                new_state = CharacterState.AWARE

        elif self.current_state == CharacterState.EXCITED:
            if self.state_timer > 5:
                new_state = CharacterState.INTERACTING

        elif self.current_state == CharacterState.SLEEPING:
            if self.detect_user_presence():
                new_state = CharacterState.AWARE

        if new_state and new_state != self.current_state:
            self.transition_to(new_state)

    def transition_to(self, new_state):
        # 退出当前状态
        self.on_state_exit(self.current_state)
        # 切换状态
        self.current_state = new_state
        self.state_timer = 0.0
        # 进入新状态
        self.on_state_enter(new_state)
```
