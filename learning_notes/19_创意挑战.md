```python
def create_context_menu(self):
    menu = QMenu(self)

    # 表情切换子菜单
    expression_menu = QMenu("切换表情", self)
    for expr in Expression:
        action = expression_menu.addAction(expr.value)
        action.triggered.connect(lambda checked, e=expr: self.switch_expression(e))
    menu.addMenu(expression_menu)

    # 设置菜单
    settings_action = menu.addAction("设置...")
    settings_action.triggered.connect(self.open_settings)

    # 关于菜单
    about_action = menu.addAction("关于")
    about_action.triggered.connect(self.show_about)

    menu.addSeparator()
    menu.addAction("退出", self.close)

    return menu
```

## 中级挑战

### 挑战 2.1：多角色系统

**难度**：⭐⭐⭐☆☆

**任务**：
扩展项目支持多个 Live2D 角色同时显示。

**要求**：
- 支持加载多个不同模型
- 实现角色排列布局（横排/纵排）
- 支持角色间切换（点击切换焦点）
- 焦点角色响应交互

**参考架构**：

```python
class MultiCharacterManager:
    def __init__(self):
        self.characters = []
        self.focus_index = 0

    def add_character(self, model_path, position):
        character = {
            "model": Live2DModel(model_path),
            "position": position,
            "controller": InteractionController()
        }
        self.characters.append(character)

    def on_click(self, pos):
        # 检测点击了哪个角色
        for i, char in enumerate(self.characters):
            if self.is_in_character(pos, char):
                self.focus_index = i
                char["controller"].handle_click(pos)
                return

    def render(self, painter):
        for i, char in enumerate(self.characters):
            # 非焦点角色缩小显示
            scale = 1.0 if i == self.focus_index else 0.7
            self.render_character(painter, char, scale)
```

### 挑战 2.2：动作序列播放

**难度**：⭐⭐⭐☆☆

**任务**：
实现动作序列功能，让角色按顺序播放多个动作。

**要求**：
- 支持定义动作序列
- 支持动作序列循环
- 添加队列管理功能
- 支持动作间的过渡效果

**参考实现**：

```python
class MotionSequence:
    def __init__(self):
        self.sequence = []
        self.current_index = 0
        self.is_playing = False
        self.transition_time = 0.3

    def add_motion(self, motion_name, duration=None):
        self.sequence.append({
            "name": motion_name,
            "duration": duration
        })

    def play(self, loop=True):
        self.is_playing = True
        self.current_index = 0
        self.play_current()

    def play_current(self):
        motion = self.sequence[self.current_index]
        self.model.play_motion(motion["name"])
        self.schedule_next(motion["duration"])

    def schedule_next(self, duration):
        # 定时播放下一个动作
        QTimer.singleShot(int(duration * 1000), self.next_motion)

    def next_motion(self):
        self.current_index += 1
        if self.current_index >= len(self.sequence):
            self.current_index = 0
            if not self.loop:
                self.is_playing = False
                return

        self.play_current()
```

将桌面宠物变成可互动游戏角色。

**要求**：
- 实现积分系统
- 实现成就系统
- 实现宠物等级系统
- 添加养成要素（喂食、换装等）

**游戏机制**：

```python
class GameSystem:
    def __init__(self):
        self.points = 0
        self.level = 1
        self.experience = 0
        self.achievements = []
        self.inventory = []

    def add_points(self, amount):
        self.points += amount
        self.check_level_up()

    def check_level_up(self):
        exp_needed = self.level * 100
        if self.experience >= exp_needed:
            self.level += 1
            self.experience = 0
            self.unlock_achievement(f"level_{self.level}")

    def unlock_achievement(self, achievement_id):
        if achievement_id not in self.achievements:
            self.achievements.append(achievement_id)
            self.show_celebration()
```

```python
class AppIntegration:
    def __init__(self):
        self.integrations = {
            "weather": WeatherAPI(),
            "calendar": CalendarAPI(),
            "email": EmailAPI(),
            "notification": NotificationListener()
        }

    def update(self):
        # 检查各应用更新
        weather = self.integrations["weather"].get_current()
        if weather:
            self.character.react_to_weather(weather)

        calendar = self.integrations["calendar"].get_upcoming()
        if calendar:
            self.remind_event(calendar)
```
