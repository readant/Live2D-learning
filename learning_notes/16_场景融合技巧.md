## 光照与阴影

### 光照匹配

角色的光照应与场景光照一致：

| 光照类型 | 阴影方向 | 应用场景 |
|----------|----------|----------|
| 自然光（上方） | 下方 | 户外场景 |
| 侧光 | 对侧 | 戏剧性效果 |
| 背光 | 轮廓光 | 艺术效果 |
| 室内光 | 下方偏前 | 室内场景 |

### 阴影实现

```python
class ShadowEffect:
    def __init__(self):
        self.enabled = True
        self.shadow_color = (0, 0, 0, 50)  # RGBA
        self.shadow_offset = (5, 10)  # 偏移量
        self.shadow_blur = 15  # 模糊程度

    def apply_shadow(self, painter, model_rect):
        """在角色下方绘制阴影"""
        if not self.enabled:
            return

        shadow_rect = QRectF(
            model_rect.x() + self.shadow_offset[0],
            model_rect.y() + model_rect.height() + self.shadow_offset[1],
            model_rect.width() * 0.8,
            10
        )

        # 创建阴影
        shadow = QGraphicsDropShadowEffect()
        shadow.setBlurRadius(self.shadow_blur)
        shadow.setColor(QColor(*self.shadow_color))
        shadow.setOffset(self.shadow_offset)

        # 应用阴影
        painter.save()
        painter.setOpacity(0.3)
        painter.fillEllipse(shadow_rect, QColor(0, 0, 0, 100))
        painter.restore()
```

### 环境光遮蔽

```python
class AmbientOcclusion:
    def __init__(self):
        self.enabled = True
        self.intensity = 0.3

    def apply_ao(self, painter, model):
        """在角色接触面添加暗角"""
        if not self.enabled:
            return

        # 底部边缘添加渐变暗色
        gradient = QRadialGradient(
            model.center_bottom,
            model.height * 0.5
        )
        gradient.setColorAt(0.8, QColor(0, 0, 0, 0))
        gradient.setColorAt(1.0, QColor(0, 0, 0, int(255 * self.intensity)))

        painter.fillRect(model.bounding_rect, gradient)
```

## 透视与层次

### 层次结构

正确的前后遮挡关系：

```
┌─────────────────────────────────┐
│  z-index: 3  背景层（最远）       │
│  ┌───────────────────────────┐  │
│  │  z-index: 2  装饰层        │  │
│  │  ┌─────────────────────┐  │  │
│  │  │  z-index: 1  角色层  │  │  │
│  │  │  ┌───────────────┐  │  │  │
│  │  │  │ z-index: 0   │  │  │  │
│  │  │  │   前景层      │  │  │  │
│  │  │  └───────────────┘  │  │  │
│  │  └─────────────────────┘  │  │
│  └───────────────────────────┘  │
└─────────────────────────────────┘
```

### 层次控制

```python
class LayerManager:
    def __init__(self):
        self.layers = {
            "background": 0,
            "decoration": 1,
            "character": 2,
            "foreground": 3,
            "effects": 4
        }

    def get_z_index(self, layer_name):
        return self.layers.get(layer_name, 0)

    def render(self, painter):
        # 按层级顺序渲染
        for layer_name in sorted(self.layers.keys(), key=lambda x: self.layers[x]):
            self.render_layer(painter, layer_name)
```

### 透视变形

根据场景透视调整角色：

```python
class PerspectiveManager:
    def __init__(self):
        self.perspective_angle = 0  # 透视角度
        self.horizontal_scale = 1.0  # 水平缩放
        self.vertical_scale = 1.0  # 垂直缩放

    def apply_perspective(self, painter, rect):
        """应用透视变形"""
        painter.save()

        # 计算透视变形矩阵
        center = rect.center()
        scale_y = self.vertical_scale
        scale_x = self.horizontal_scale

        # 应用变换
        transform = QTransform()
        transform.translate(center.x(), center.y())
        transform.scale(scale_x, scale_y)
        transform.translate(-center.x(), -center.y())

        painter.setTransform(transform)
        painter.restore()
```

### 动态背景元素

```python
class DynamicBackground:
    def __init__(self):
        self.elements = []

    def add_element(self, element_type, config):
        if element_type == "particle":
            self.elements.append(ParticleElement(config))
        elif element_type == "wave":
            self.elements.append(WaveElement(config))
        elif element_type == "cloud":
            self.elements.append(CloudElement(config))

    def update(self, delta_time):
        for element in self.elements:
            element.update(delta_time)

    def render(self, painter):
        for element in self.elements:
            element.render(painter)
```

### 粒子背景

```python
class ParticleElement:
    def __init__(self, config):
        self.particles = []
        self.max_particles = config.get("max", 50)
        self.particle_life = config.get("life", 5.0)
        self.gravity = config.get("gravity", -10)
        self.wind = config.get("wind", 0)

    def emit(self, pos, velocity):
        if len(self.particles) < self.max_particles:
            self.particles.append({
                "x": pos[0],
                "y": pos[1],
                "vx": velocity[0],
                "vy": velocity[1],
                "life": self.particle_life,
                "size": random.uniform(2, 5)
            })

    def update(self, delta_time):
        for p in self.particles[:]:
            p["x"] += p["vx"] * delta_time
            p["y"] += p["vy"] * delta_time
            p["vy"] += self.gravity * delta_time
            p["vx"] += self.wind * delta_time
            p["life"] -= delta_time

            if p["life"] <= 0:
                self.particles.remove(p)

    def render(self, painter):
        for p in self.particles:
            alpha = int(255 * (p["life"] / self.particle_life))
            painter.setOpacity(alpha / 255)
            painter.fillRect(
                p["x"], p["y"], p["size"], p["size"],
                QColor(255, 255, 255, alpha)
            )
```

### 天气效果

```python
class WeatherEffect:
    def __init__(self):
        self.type = "none"  # none, rain, snow, wind
        self.intensity = 0.0  # 0.0 ~ 1.0
        self.particles = RainParticles() if self.type == "rain" else None

    def set_weather(self, weather_type, intensity):
        self.type = weather_type
        self.intensity = intensity

        if weather_type == "rain":
            self.particles = RainParticles(intensity)
        elif weather_type == "snow":
            self.particles = SnowParticles(intensity)

    def update(self, delta_time):
        if self.particles:
            self.particles.update(delta_time)

    def render(self, painter):
        if self.particles:
            self.particles.render(painter)
```

## 多角色组合

### 多角色管理

```python
class MultiCharacterManager:
    def __init__(self):
        self.characters = []
        self.arrangement = "horizontal"  # horizontal, vertical, formation

    def add_character(self, character, position=None):
        self.characters.append({
            "character": character,
            "position": position or self.get_default_position(len(self.characters)),
            "z_index": len(self.characters)
        })

    def get_default_position(self, index):
        if self.arrangement == "horizontal":
            x = 100 + index * 150
            y = 100
        elif self.arrangement == "vertical":
            x = 100
            y = 100 + index * 200
        else:  # formation
            positions = [(100, 100), (200, 150), (150, 200)]
            return positions[index % len(positions)]

        return (x, y)

    def update(self, delta_time):
        for char_data in self.characters:
            char_data["character"].update(delta_time)

    def render(self, painter):
        # 按 z_index 排序
        sorted_chars = sorted(self.characters, key=lambda x: x["z_index"])

        for char_data in sorted_chars:
            painter.save()
            painter.translate(*char_data["position"])
            char_data["character"].render(painter)
            painter.restore()
```

### 角色间交互

```python
class CharacterInteraction:
    def __init__(self, characters):
        self.characters = characters

    def update(self, delta_time):
        for i, char1 in enumerate(self.characters):
            for j, char2 in enumerate(self.characters):
                if i != j:
                    self.handle_interaction(char1, char2)

    def handle_interaction(self, char1, char2):
        distance = self.get_distance(char1.position, char2.position)

        # 角色间的视线跟随
        if distance < 300:
            char1.look_at(char2.position)
            char2.look_at(char1.position)

        # 角色间保持最小距离
        min_distance = 80
        if distance < min_distance:
            self.separate_characters(char1, char2, min_distance)
```

## 场景融合示例

### 桌面宠物场景

```python
class DesktopScene:
    def __init__(self):
        self.background = None
        self.character = None
        self.effects = []
        self.layer_manager = LayerManager()

    def setup_desktop_scene(self):
        # 添加模糊桌面背景
        self.background = BlurredDesktopBackground(blur_radius=10)

        # 添加角色
        self.character = Live2DCharacter()

        # 添加粒子效果
        self.add_effect("sparkle", {
            "max_particles": 20,
            "emission_rate": 2,
            "colors": [(255, 215, 0), (255, 255, 255)]
        })

    def update(self, delta_time):
        self.background.update(delta_time)
        self.character.update(delta_time)

        for effect in self.effects:
            effect.update(delta_time)

    def render(self, painter):
        # 按层级渲染
        self.background.render(painter)  # 背景层
        self.character.render(painter)   # 角色层

        for effect in self.effects:      # 效果层
            effect.render(painter)
```
