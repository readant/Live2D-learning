### 核心代码实现

#### 1. 模型加载

```python
class Live2DModel:
    def __init__(self, model_path):
        self.model = None
        self.motions = {}
        self.expressions = {}
        self.load_model(model_path)

    def load_model(self, model_path):
        # 加载 moc 文件
        self.model = Live2DModelWrapper(model_path)

        # 加载动作
        motion_dir = os.path.join(model_path, "motion")
        for motion_file in os.listdir(motion_dir):
            if motion_file.endswith(".motion3.json"):
                motion_name = motion_file.replace(".motion3.json", "")
                self.motions[motion_name] = self.load_motion(
                    os.path.join(motion_dir, motion_file)
                )

        # 加载表情
        expression_dir = os.path.join(model_path, "expression")
        for expr_file in os.listdir(expression_dir):
            if expr_file.endswith(".expression.json"):
                expr_name = expr_file.replace(".expression.json", "")
                self.expressions[expr_name] = self.load_expression(
                    os.path.join(expression_dir, expr_file)
                )

    def play_motion(self, motion_name, loop=False):
        if motion_name in self.motions:
            self.model.start_motion(self.motions[motion_name], loop=loop)

    def set_expression(self, expression_name):
        if expression_name in self.expressions:
            self.model.set_expression(self.expressions[expression_name])
```

#### 2. 交互处理

```python
class InteractionController:
    def __init__(self, model):
        self.model = model
        self.click_count = 0
        self.click_timer = 0.0
        self.last_click_time = 0

    def handle_click(self, pos, timestamp):
        # 检测连击
        if timestamp - self.last_click_time < 0.3:
            self.click_count += 1
        else:
            self.click_count = 1

        self.last_click_time = timestamp

        # 根据连击数触发不同动作
        if self.click_count == 1:
            self.on_single_click(pos)
        elif self.click_count == 2:
            self.on_double_click(pos)
        elif self.click_count >= 3:
            self.on_triple_click(pos)

    def on_single_click(self, pos):
        # 头部或身体点击
        if self.is_head_area(pos):
            self.model.play_motion("TapHead")
            self.model.set_expression("Happy")
        else:
            self.model.play_motion("TapBody")

    def on_double_click(self, pos):
        self.model.play_motion("Flick")

    def on_triple_click(self, pos):
        self.model.play_motion("FlickDown")
```

#### 3. 动画系统

```python
class AnimationController:
    def __init__(self, model):
        self.model = model
        self.active_motions = {}
        self.idle_motion = "Idle"

    def update(self, delta_time):
        # 更新主动画
        for motion_name, motion_data in self.active_motions.items():
            motion_data["elapsed"] += delta_time

            # 检查是否播放完成
            if motion_data["elapsed"] >= motion_data["duration"]:
                if not motion_data["loop"]:
                    self.stop_motion(motion_name)
                else:
                    motion_data["elapsed"] = 0

        # 播放待机动画
        if not self.active_motions:
            self.model.play_motion(self.idle_motion, loop=True)

    def play_motion(self, motion_name, loop=False, priority=0):
        # 优先级检查
        for existing, data in self.active_motions.items():
            if data["priority"] > priority:
                return False

        self.model.play_motion(motion_name, loop=loop)
        self.active_motions[motion_name] = {
            "priority": priority,
            "elapsed": 0,
            "duration": self.get_motion_duration(motion_name),
            "loop": loop
        }
        return True

    def stop_motion(self, motion_name):
        if motion_name in self.active_motions:
            self.model.stop_motion(motion_name)
            del self.active_motions[motion_name]
```

```python
class DanmakuInteraction:
    def __init__(self, character):
        self.character = character
        self.ws_client = None
        self.danmaku_queue = Queue()

    def connect_to_danmaku(self, room_id):
        self.ws_client = WebSocketClient(
            f"wss://api.douyu.com/{room_id}"
        )
        self.ws_client.on_message = self.on_danmaku_received

    def on_danmaku_received(self, danmaku_data):
        # 解析弹幕
        text = danmaku_data.get("text", "")
        user = danmaku_data.get("user", "")

        # 分析弹幕意图
        intent = self.analyze_intent(text)

        # 生成响应
        response = self.generate_response(intent, user)
        self.danmaku_queue.put(response)

    def analyze_intent(self, text):
        keywords = {
            "greeting": ["你好", "哈喽", "早上好", "晚上好"],
            "praise": ["可爱", "好看", "厉害", "棒"],
            "question": ["？", "怎么", "什么", "为什么"],
            "command": ["跳", "笑", "哭", "生气"]
        }

        for intent, words in keywords.items():
            if any(word in text for word in words):
                return intent
        return "unknown"

    def generate_response(self, intent, user):
        responses = {
            "greeting": {
                "action": "Wave",
                "expression": "Happy",
                "message": f"大家好，我是{user}的虚拟助手！"
            },
            "praise": {
                "action": "Jump",
                "expression": "Happy",
                "message": "谢谢夸奖！"
            },
            "question": {
                "action": "Thinking",
                "expression": "Serious",
                "message": "让我想想..."
            },
            "command": {
                "action": "Jump",
                "expression": "Happy",
                "message": "好的！"
            }
        }

        return responses.get(intent, responses["unknown"])
```

```python
class LipSync:
    def __init__(self):
        self.audio_analyzer = AudioAnalyzer()
        self.mouth_open = 0.0

    def update_from_audio(self, audio_data):
        # 分析音频
        volume = self.audio_analyzer.get_volume(audio_data)
        frequency = self.audio_analyzer.get_main_frequency(audio_data)

        # 简单口型同步：音量越大嘴巴越开
        target_mouth = min(1.0, volume * 2)
        self.mouth_open += (target_mouth - self.mouth_open) * 0.3

        return self.mouth_open

    def apply_to_model(self, model):
        model.set_parameter("ParamMouthOpenY", self.mouth_open)
```

#### 3. 动作队列

```python
class ActionQueue:
    def __init__(self, character):
        self.character = character
        self.queue = deque()
        self.current_action = None
        self.blend_time = 0.3

    def enqueue(self, action, expression=None, audio=None):
        self.queue.append({
            "action": action,
            "expression": expression,
            "audio": audio
        })

    def update(self, delta_time):
        if not self.current_action:
            # 取出下一个动作
            if self.queue:
                self.current_action = self.queue.popleft()
                self.execute_action(self.current_action)
                self.action_elapsed = 0.0
        else:
            self.action_elapsed += delta_time

            # 动作完成后检查队列
            if self.action_elapsed >= self.current_action["duration"]:
                self.current_action = None

    def execute_action(self, action_data):
        # 播放动作
        if action_data["action"]:
            self.character.play_motion(action_data["action"])

        # 切换表情
        if action_data["expression"]:
            self.character.set_expression(action_data["expression"])

        # 播放音频
        if action_data["audio"]:
            self.character.play_audio(action_data["audio"])
```

```python
class QA_system:
    def __init__(self, knowledge_base):
        self.knowledge_base = knowledge_base
        self.conversation_history = []

    def ask(self, question):
        # 理解问题
        intent = self.nlu.parse(question)

        # 检索答案
        answer = self.knowledge_base.search(intent)

        # 生成响应
        response = self.generate_response(intent, answer)

        # 记录对话历史
        self.conversation_history.append({
            "question": question,
            "answer": answer,
            "intent": intent,
            "timestamp": time.time()
        })

        return response

    def generate_response(self, intent, answer):
        emotion = self.analyze_emotion(intent)

        return {
            "text": answer,
            "emotion": emotion,
            "action": self.emotion_to_action(emotion)
        }

    def emotion_to_action(self, emotion):
        mapping = {
            "happy": {"action": "Happy", "expression": "Happy"},
            "thinking": {"action": "Thinking", "expression": "Serious"},
            "sad": {"action": "Sad", "expression": "Sad"},
            "excited": {"action": "Jump", "expression": "Surprised"}
        }
        return mapping.get(emotion, mapping["thinking"])
```

#### 2. 学习进度跟踪

```python
class LearningTracker:
    def __init__(self):
        self.user_progress = {}
        self.daily_goals = {
            "lessons": 2,
            "exercises": 5,
            "review": 3
        }

    def record_activity(self, user_id, activity_type):
        if user_id not in self.user_progress:
            self.user_progress[user_id] = {
                "lessons_completed": 0,
                "exercises_completed": 0,
                "reviews_completed": 0,
                "total_time": 0,
                "streak_days": 0
            }

        progress = self.user_progress[user_id]

        if activity_type == "lesson":
            progress["lessons_completed"] += 1
        elif activity_type == "exercise":
            progress["exercises_completed"] += 1
        elif activity_type == "review":
            progress["reviews_completed"] += 1

        # 检查目标达成
        self.check_goals(user_id)

    def get_progress_report(self, user_id):
        progress = self.user_progress.get(user_id, {})
        goals = self.daily_goals

        return {
            "lessons": f"{progress.get('lessons_completed', 0)}/{goals['lessons']}",
            "exercises": f"{progress.get('exercises_completed', 0)}/{goals['exercises']}",
            "reviews": f"{progress.get('reviews_completed', 0)}/{goals['review']}",
            "streak": progress.get('streak_days', 0)
        }

    def check_goals(self, user_id):
        progress = self.user_progress[user_id]
        goals = self.daily_goals

        all_complete = (
            progress["lessons_completed"] >= goals["lessons"] and
            progress["exercises_completed"] >= goals["exercises"] and
            progress["reviews_completed"] >= goals["review"]
        )

        if all_complete:
            # 触发庆祝动画
            return True
        return False
```
